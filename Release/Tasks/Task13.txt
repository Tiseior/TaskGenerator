  Реализовать сортировку выбором.
  Ищем наименьшее значение в массиве и ставим его на позицию,
откуда начали проход. Потом двигаемся на следующую позицию.

  Пример: 12, 5, 48, 0, 4
1) 12, 5, 48, (0), 4
    0, 12, 48, 5, 4
2) 0, 12, 48, 5, (4)
    0, 4, 48, 5, 12
3) 0, 4, 48, (5), 12
    0, 4, 5, 48, 12
4) 0, 4, 5, 48, (12)
    0, 4, 5, 12, 48
