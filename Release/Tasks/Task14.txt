  Реализовать сортировку вставками.
  Начинаем со второго элемента. Проверяем между собой второй
элемент с первым и, если надо, меняем местами. Сравниваем 
следующую пару элементов и проверяем все пары до неё.

  Пример: 12, 5, 48, 0, 4
1) (12, 5), 48, 0, 4
    (5, 12), 48, 0, 4
2) 5, (12, 48), 0, 4
    (5, 12), 48, 0, 4
3) 5, 12, 48, (0, 4)
    5, 12, (48, 0), 4
    5, 12, (0, 48), 4
    5, (12, 0), 48, 4
    5, (0, 12), 48, 4
    (5, 0), 12, 48, 4
    0, 5, 12, 48, 4
4) 0, 5, 12, (48, 4)
    0, 5, 12, (4, 48)
    0, 5, (12, 4), 48
    0, 5, (4, 12), 48
    0, (5, 4), 12, 48
    0, (4, 5), 12, 48
    (0, 4), 5, 12, 48
    0, 4, 5, 12, 48
